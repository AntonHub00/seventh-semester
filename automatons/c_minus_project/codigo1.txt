/*
* A program to perform Euclid's algorithm to computer
* the greates common denominator (GCD) of a pair of integers.
*/

int gcd(int u, int v)
{
	if(v==0) return u;
	else return gcd(v,u-u/v*v, v+v, u*u);
	/* u-u/v*v == u mod v */
}
